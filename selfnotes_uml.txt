class App {

class Main {
  -scanner: Scanner
  -manager: NoteManager

  +main(String[] args)
}

class User {
  -username: String
  +getUsername(): String
}

class Note {
  -id: int
  -title: String
  -subject: String
  -uploaderName: String
  -content: String
  -filePath: String
  -fileType: String
  -state: NoteState
  +getId(): int
  +getTitle(): String
  +getSubject(): String
  +getUploaderName(): String
  +getFileType(): String
  +isOwner(user: User): boolean
  +changeState(nextState: NoteState): void
  +getStateName(): String
  +read(viewer: User): String
  +getDisplayContent(): String
}

class NoteContentData {
  -content: String
  -filePath: String
  -fileType: String
  +getContent(): String
  +getFilePath(): String
  +getFileType(): String
}

class NoteManager <<Singleton>> {
  -instance: NoteManager
  -notes: List<Note>
  -observers: List<NoteObserver>
  -noteFactory: NoteFactory
  +getInstance(): NoteManager
  +addObserver(observer: NoteObserver): void
  +addNote(user: User, subject: String, title: String, data: NoteContentData): Note
  +changeNoteState(user: User, noteId: int, nextState: NoteState): boolean
  +getSubjects(): List<String>
  +getNotesBySubject(subject: String): List<Note>
}

class NoteFactory <<Factory>> {
  +createNote(id: int, title: String, subject: String, uploaderName: String, content: String, filePath: String, fileType: String): Note
}

interface NoteObserver <<Observer>> {
  +update(action: String, note: Note): void
}

class NotificationService {
  +update(action: String, note: Note): void
}

interface NoteCreationStrategy <<Strategy>> {
  +createContent(): NoteContentData
}

class ManualNoteStrategy {
  -text: String
  +createContent(): NoteContentData
}

class FileNoteStrategy {
  -filePath: String
  +createContent(): NoteContentData
}

class NoteCreatorContext {
  -strategy: NoteCreationStrategy
  +setStrategy(strategy: NoteCreationStrategy): void
  +createContent(): NoteContentData
}

interface NoteState <<State>> {
  +getName(): String
  +read(note: Note, viewer: User): String
}

class DraftState {
  +getName(): String
  +read(note: Note, viewer: User): String
}

class PublishedState {
  +getName(): String
  +read(note: Note, viewer: User): String
}

class ArchivedState {
  +getName(): String
  +read(note: Note, viewer: User): String
}
